 34 "use strict";
 35 (function($) {
 36 
 37     var pluginName = 'jui_pagination';
 38 
 39     /* public methods ------------------------------------------------------- */
 40     var methods = {
 41 
 42         /**
 43          * @lends $.fn.jui_pagination
 44          */
 45         init: function(options) {
 46 
 47             var elem = this;
 48 
 49             return this.each(function() {
 50 
 51                 /**
 52                  * settings and defaults
 53                  * using $.extend, settings modification will affect elem.data() and vive versa
 54                  */
 55                 var settings = elem.data(pluginName);
 56                 if(typeof(settings) == 'undefined') {
 57                     var defaults = elem.jui_pagination('getDefaults');
 58                     settings = $.extend({}, defaults, options);
 59                 } else {
 60                     settings = $.extend({}, settings, options);
 61                 }
 62                 elem.data(pluginName, settings);
 63 
 64                 var container_id = elem.attr("id");
 65 
 66                 // simple validation
 67                 validate_input(container_id);
 68 
 69                 if(elem.data('error_occured')) {
 70                     elem.html('');
 71                     elem.data('error_occured', false);
 72                 }
 73 
 74                 // bind events
 75                 elem.unbind("onChangePage").bind("onChangePage", settings.onChangePage);
 76                 elem.unbind("onSetRowsPerPage").bind("onSetRowsPerPage", settings.onSetRowsPerPage);
 77                 elem.unbind("onDisplay").bind("onDisplay", settings.onDisplay);
 78 
 79                 // retrieve options
 80                 var totalPages = settings.totalPages;
 81                 var currentPage = settings.currentPage;
 82                 var visiblePageLinks = settings.visiblePageLinks;
 83                 var rowsPerPage = settings.rowsPerPage;
 84 
 85                 var useSlider = settings.useSlider;
 86                 var sliderElementID = settings.sliderElementID;
 87                 var useSliderWithPagesCount = settings.useSliderWithPagesCount;
 88                 var sliderOrientation = settings.sliderOrientation;
 89                 var sliderAnimation = settings.sliderAnimation;
 90                 var sliderInsidePane = settings.sliderInsidePane;
 91 
 92                 var showGoToPage = settings.showGoToPage;
 93                 var showNavButtons = settings.showNavButtons;
 94                 var showLabelCurrentPage = settings.showLabelCurrentPage;
 95                 var showCurrentPage = settings.showCurrentPage;
 96                 var showNavPages = settings.showNavPages;
 97                 var navPagesMode = settings.navPagesMode;
 98                 var showLabelTotalPages = settings.showLabelTotalPages;
 99                 var showTotalPages = settings.showTotalPages;
100                 var showRowsPerPage = settings.showRowsPerPage;
101                 var showRowsInfo = settings.showRowsInfo;
102                 var showPreferences = settings.showPreferences;
103 
104                 var nav_pane_display_order = settings.nav_pane_display_order;
105 
106                 var containerClass = settings.containerClass;
107                 var navPaneClass = settings.navPaneClass;
108                 var navGoToPageClass = settings.navGoToPageClass;
109                 var navButtonTopClass = settings.navButtonTopClass;
110                 var navButtonPrevClass = settings.navButtonPrevClass;
111                 var navCurrentPageLabelClass = settings.navCurrentPageLabelClass;
112                 var navCurrentPageClass = settings.navCurrentPageClass;
113                 var navPagesClass = settings.navPagesClass;
114                 var navTotalPagesLabelClass = settings.navTotalPagesLabelClass;
115                 var navTotalPagesClass = settings.navTotalPagesClass;
116                 var navButtonNextClass = settings.navButtonNextClass;
117                 var navButtonLastClass = settings.navButtonLastClass;
118                 var navRowsPerPageClass = settings.navRowsPerPageClass;
119                 var navInfoClass = settings.navInfoClass;
120                 var preferencesClass = settings.preferencesClass;
121                 var sliderClass = (settings.sliderInsidePane ? settings.sliderInsidePaneClass : settings.sliderClass);
122 
123                 var nav_pane_id = create_id(settings.nav_pane_id_prefix, container_id);
124                 var goto_page_id = create_id(settings.nav_goto_page_id_prefix, container_id);
125                 var current_label_id = create_id(settings.nav_current_page_label_id_prefix, container_id);
126                 var current_id = create_id(settings.nav_current_page_id_prefix, container_id);
127                 var nav_top_id = create_id(settings.nav_top_id_prefix, container_id);
128                 var nav_prev_id = create_id(settings.nav_prev_id_prefix, container_id);
129                 var nav_pages_id = create_id(settings.nav_pages_id_prefix, container_id);
130                 var nav_item_id_prefix = create_id(settings.nav_item_id_prefix, container_id) + '_';
131                 var nav_next_id = create_id(settings.nav_next_id_prefix, container_id);
132                 var nav_last_id = create_id(settings.nav_last_id_prefix, container_id);
133                 var total_label_id = create_id(settings.nav_total_pages_label_id_prefix, container_id);
134                 var total_id = create_id(settings.nav_total_pages_id_prefix, container_id);
135                 var rows_per_page_id = create_id(settings.nav_rows_per_page_id_prefix, container_id);
136                 var rows_info_id = create_id(settings.nav_rows_info_id_prefix, container_id);
137                 var preferences_id = create_id(settings.preferences_id_prefix, container_id);
138                 var pref_dialog_id = create_id(settings.pref_dialog_id_prefix, container_id);
139                 var slider_id = (!sliderElementID ? create_id(settings.slider_id_prefix, container_id) : sliderElementID);
140 
141                 var disableSelectionNavPane = settings.disableSelectionNavPane;
142 
143                 var elem_nav_pane;
144                 var elem_slider;
145                 var selector;
146 
147                 if(useSlider) {
148                     var pageLimit = (useSliderWithPagesCount == 0 ? visiblePageLinks : Math.max(useSliderWithPagesCount, visiblePageLinks));
149                     if(totalPages <= pageLimit) {
150                         useSlider = false;
151                     }
152                 }
153 
154                 // set container style
155                 if(containerClass != '') {
156                     elem.removeClass().addClass(containerClass);
157                 }
158 
159                 /* CREATE PANEL --------------------------------------------- */
160                 if($("#" + nav_pane_id).length == 0) {
161                     elem.html('<div id="' + nav_pane_id + '"></div>' + elem.html());
162                 }
163                 elem_nav_pane = $("#" + nav_pane_id);
164 
165                 var nav_pane_html = '';
166 
167                 if(showPreferences) {
168                     nav_pane_html += '<div id="' + preferences_id +
169                         '" title="' + rsc_jui_pag.preferences_title + '">'
170                         + rsc_jui_pag.preferences_text + '</div>';
171                 }
172 
173                 $.each(nav_pane_display_order, function(index, value) {
174 
175                     if(value == 'go_to_page') {
176                         if(showGoToPage) {
177                             nav_pane_html += '<input type="text" id="' + goto_page_id + '" title="' + rsc_jui_pag.go_to_page_title + '">';
178                         }
179                     } else if(value == 'slider') {
180                         if(useSlider && sliderInsidePane) {
181                             elem_slider = $("#" + slider_id);
182                             if(typeof(elem_slider.data("slider")) == 'object') {
183                                 elem_slider.slider('destroy');
184                                 elem_slider.html('');
185                             }
186                             nav_pane_html += '<div id="' + slider_id + '"></div>';
187                         }
188                     } else if(value == 'back_buttons') {
189                         if(showNavButtons) {
190                             nav_pane_html += '<div id="' + nav_top_id + '">' + rsc_jui_pag.go_top_text + '</div>';
191                             nav_pane_html += '<div id="' + nav_prev_id + '">' + rsc_jui_pag.go_prev_text + '</div>';
192                         }
193                     } else if(value == 'current_page_label') {
194                         if(showLabelCurrentPage) {
195                             nav_pane_html += '<div id="' + current_label_id + '">' + rsc_jui_pag.current_page_label + '</div>';
196                         }
197                     } else if(value == 'current_page') {
198                         if(showCurrentPage) {
199                             nav_pane_html += '<div id="' + current_id + '">' + currentPage + '</div>';
200                         }
201                     } else if(value == 'nav_items') {
202                         if(showNavPages) {
203                             nav_pane_html += '<div id="' + nav_pages_id + '"></div>';
204                         }
205                     } else if(value == 'total_pages_label') {
206                         if(showLabelTotalPages) {
207                             nav_pane_html += '<div id="' + total_label_id + '">' + rsc_jui_pag.total_pages_label + '</div>';
208                         }
209                     } else if(value == 'total_pages') {
210                         if(showTotalPages) {
211                             nav_pane_html += '<div id="' + total_id + '">' + totalPages + '</div>';
212                         }
213                     } else if(value == 'forward_buttons') {
214                         if(showNavButtons) {
215                             nav_pane_html += '<div id="' + nav_next_id + '">' + rsc_jui_pag.go_next_text + '</div>';
216                             nav_pane_html += '<div id="' + nav_last_id + '">' + rsc_jui_pag.go_last_text + '</div>';
217                         }
218                     } else if(value == 'rows_per_page') {
219                         if(showRowsPerPage) {
220                             nav_pane_html += '<input type="text" id="' + rows_per_page_id + '" ' +
221                                 'value="' + rowsPerPage + '" ' +
222                                 'title="' + rsc_jui_pag.rows_per_page_title + '">';
223                         }
224                     } else if(value == 'rows_info') {
225                         if(showRowsInfo) {
226                             nav_pane_html += '<div id="' + rows_info_id + '"></div>';
227                         }
228                     }
229 
230                 });
231 
232                 // set nav_pane_html
233                 elem_nav_pane.html(nav_pane_html);
234 
235                 // apply style
236                 elem_nav_pane.removeClass().addClass(navPaneClass);
237 
238                 $("#" + goto_page_id).removeClass().addClass(navGoToPageClass);
239 
240                 $("#" + current_label_id).removeClass().addClass(navCurrentPageLabelClass);
241                 $("#" + current_id).removeClass().addClass(navCurrentPageClass);
242 
243                 $("#" + nav_top_id).removeClass().addClass(navButtonTopClass);
244                 $("#" + nav_prev_id).removeClass().addClass(navButtonPrevClass);
245 
246                 $("#" + nav_pages_id).removeClass().addClass(navPagesClass);
247 
248                 $("#" + nav_next_id).removeClass().addClass(navButtonNextClass);
249                 $("#" + nav_last_id).removeClass().addClass(navButtonLastClass);
250 
251                 $("#" + total_label_id).removeClass().addClass(navTotalPagesLabelClass);
252                 $("#" + total_id).removeClass().addClass(navTotalPagesClass);
253 
254                 $("#" + rows_per_page_id).removeClass().addClass(navRowsPerPageClass);
255 
256                 $("#" + rows_info_id).removeClass().addClass(navInfoClass);
257 
258                 $("#" + preferences_id).removeClass().addClass(preferencesClass);
259 
260                 create_nav_items(container_id);
261 
262                 // panel enents
263                 if(showNavButtons) {
264 
265                     // click on go to top button
266                     selector = "#" + nav_top_id;
267                     elem_nav_pane.off('click', selector).on('click', selector, function() {
268                         settings.currentPage = 1;
269                         change_page(container_id, settings.currentPage, true, true);
270                     });
271 
272                     // click on go to prev button
273                     selector = "#" + nav_prev_id;
274                     elem_nav_pane.off('click', selector).on('click', selector, function() {
275                         if(settings.currentPage > 1) {
276                             settings.currentPage = parseInt(settings.currentPage) - 1;
277                             change_page(container_id, settings.currentPage, true, true);
278                         }
279                     });
280 
281                     // click on go to next button
282                     selector = "#" + nav_next_id;
283                     elem_nav_pane.off('click', selector).on('click', selector, function() {
284                         if(settings.currentPage < settings.totalPages) {
285                             settings.currentPage = parseInt(settings.currentPage) + 1;
286                             change_page(container_id, settings.currentPage, true, true);
287                         }
288                     });
289 
290                     // click on go to end button
291                     selector = "#" + nav_last_id;
292                     elem_nav_pane.off('click', selector).on('click', selector, function() {
293                         settings.currentPage = parseInt(settings.totalPages);
294                         change_page(container_id, settings.currentPage, true, true);
295                     });
296 
297                 }
298 
299                 // click on nav page item
300                 selector = '[id^="' + nav_item_id_prefix + '"]';
301                 elem_nav_pane.off('click', selector).on('click', selector, function(event) {
302                     var len = nav_item_id_prefix.length;
303                     settings.currentPage = parseInt($(event.target).attr("id").substr(len));
304                     if(navPagesMode == 'continuous') {
305                         change_page(container_id, settings.currentPage, false, true);
306                     } else if(navPagesMode == 'first-last-always-visible') {
307                         change_page(container_id, settings.currentPage, true, true);
308                     }
309                 });
310 
311                 // go to page event
312                 if(showGoToPage) {
313                     selector = "#" + goto_page_id;
314                     elem_nav_pane.off('keypress', selector).on('keypress', selector, function(event) {
315                         if(event.which === 13) {
316                             var gtp = parseInt($(event.target).val());
317                             $("#" + goto_page_id).val('');
318                             if(!isNaN(gtp) && gtp > 0) {
319                                 settings.currentPage = gtp;
320                                 if(settings.currentPage > settings.totalPages) {
321                                     settings.currentPage = settings.totalPages;
322                                 }
323                                 change_page(container_id, settings.currentPage, true, true);
324                             } else {
325                                 elem.triggerHandler("onChangePage", settings.currentPage);
326                             }
327                             event.preventDefault(); // MSIE strange behaviour
328                         } else {
329                             if(!(event.which === 8 || event.which === 0 || (event.shiftKey === false && (event.which > 47 && event.which < 58)))) {
330                                 event.preventDefault();
331                             }
332                         }
333                     });
334                 }
335 
336                 // rows per page event
337                 if(showRowsPerPage) {
338                     selector = "#" + rows_per_page_id;
339                     elem_nav_pane.off('keypress', selector).on('keypress', selector, function(event) {
340                         if(event.which === 13) {
341                             var rpp = parseInt($(event.target).val());
342                             if(!isNaN(rpp) && rpp > 0) {
343                                 settings.rowsPerPage = rpp;
344                             } else {
345                                 $("#" + rows_per_page_id).val(settings.rowsPerPage);
346                             }
347                             elem.triggerHandler("onSetRowsPerPage", rpp);
348                             event.preventDefault(); // MSIE strange behaviour
349                         } else {
350                             if(!(event.which === 8 || event.which === 0 || (event.shiftKey === false && (event.which > 47 && event.which < 58)))) {
351                                 event.preventDefault();
352                             }
353                         }
354                     });
355                 }
356 
357 
358                 if(disableSelectionNavPane) {
359                     disableSelection(elem_nav_pane);
360                 }
361 
362 
363                 /* CREATE SLIDER -------------------------------------------- */
364                 if(useSlider) {
365 
366                     if(!sliderElementID && !sliderInsidePane) {
367                         if($("#" + slider_id).length == 0) {
368                             elem.append('<div id="' + slider_id + '"></div>');
369                         }
370                     }
371                     elem_slider = $("#" + slider_id);
372 
373                     if(sliderElementID) {
374                         selector = "#" + create_id(settings.slider_id_prefix, container_id);
375                         if(typeof($(selector).data("slider")) == 'object') {
376                             $(selector).slider('destroy');
377                             $(selector).html('');
378                         }
379                     }
380 
381                     elem_slider.removeClass(sliderClass).addClass(sliderClass);
382 
383                     elem_slider.slider({
384                         min: 1,
385                         max: totalPages,
386                         value: (sliderOrientation == 'horizontal' ? currentPage : totalPages - currentPage + 1),
387                         animate: sliderAnimation,
388                         range: (sliderOrientation == 'horizontal' ? 'min' : 'max'),
389                         orientation: sliderOrientation,
390                         stop: function(event, ui) {
391                             settings.currentPage = (sliderOrientation == 'horizontal' ? parseInt(ui.value) : parseInt(totalPages) - parseInt(ui.value) + 1);
392                             change_page(container_id, settings.currentPage, true, false);
393                         }
394                     });
395 
396                 } else {
397                     elem_slider = $("#" + slider_id);
398                     if(typeof(elem_slider.data("slider")) == 'object') {
399                         elem_slider.slider('destroy');
400                         elem_slider.html('');
401                     }
402                 }
403 
404                 /* CREATE PREFERENCES DIALOG -------------------------------- */
405                 if(showPreferences) {
406 
407                     if($('#' + pref_dialog_id).length == 0) {
408                         var pref_html = '<div id="' + pref_dialog_id + '"></div>';
409                         elem.append(pref_html);
410                     }
411 
412                     selector = "#" + preferences_id;
413                     var elem_pref_dialog = $("#" + pref_dialog_id);
414 
415                     elem_nav_pane.off('click', selector).on('click', selector, function() {
416 
417                         if(typeof(elem_pref_dialog.data('dialog')) == 'object') {
418                             elem_pref_dialog.dialog('destroy');
419                         }
420 
421                         elem_pref_dialog.dialog({
422                             autoOpen: true,
423                             show: "blind",
424                             hide: "explode",
425                             position: {
426                                 my: "top",
427                                 at: "bottom",
428                                 of: '#' + container_id
429                             },
430                             title: rsc_jui_pag.preferences_title,
431                             buttons: [
432                                 {
433                                     text: rsc_jui_pag.preferences_close,
434                                     click: function() {
435                                         $(this).dialog("close");
436                                         $(this).dialog("destroy");
437                                     }
438                                 }
439                             ],
440                             open: create_preferences(container_id)
441                         })
442                     });
443 
444                     selector = "#" + pref_dialog_id + '_slider';
445                     elem_pref_dialog.off('click', selector).on('click', selector, function(event) {
446                         var state = $(event.target).is(":checked");
447                         elem.jui_pagination({
448                             useSlider: state
449                         })
450                     });
451 
452                     selector = "#" + pref_dialog_id + '_goto_page';
453                     elem_pref_dialog.off('click', selector).on('click', selector, function(event) {
454                         var state = $(event.target).is(":checked");
455                         elem.jui_pagination({
456                             showGoToPage: state
457                         })
458                     });
459 
460                     selector = "#" + pref_dialog_id + '_rows_per_page';
461                     elem_pref_dialog.off('click', selector).on('click', selector, function(event) {
462                         var state = $(event.target).is(":checked");
463                         elem.jui_pagination({
464                             showRowsPerPage: state
465                         })
466                     });
467 
468                     selector = "#" + pref_dialog_id + '_rows_info';
469                     elem_pref_dialog.off('click', selector).on('click', selector, function(event) {
470                         var state = $(event.target).is(":checked");
471                         elem.jui_pagination({
472                             showRowsInfo: state
473                         })
474                     });
475 
476                     selector = "#" + pref_dialog_id + '_nav_buttons';
477                     elem_pref_dialog.off('click', selector).on('click', selector, function(event) {
478                         var state = $(event.target).is(":checked");
479                         elem.jui_pagination({
480                             showNavButtons: state
481                         })
482                     });
483 
484                 }
485 
486                 elem.triggerHandler('onDisplay');
487 
488             });

707             elem.html('<span style="color: red;">' + 'ERROR: ' + err_msg + '</span>');
708             elem.data('error_occured', true);

783     /**
784      * Create nagivation pages
785      * @param container_id
786      */
787     var create_nav_items = function(container_id) {
788 
789         // retrieve options
790         var elem = $("#" + container_id);
791         var s = $(elem).jui_pagination('getAllOptions');
792 
793         var navPagesMode = s.navPagesMode;
794 
795         var totalPages = s.totalPages;
796         var currentPage = s.currentPage;
797         var visiblePageLinks = s.visiblePageLinks;
798 
799         var showNavPages = s.showNavPages;
800         var showNavButtons = s.showNavButtons;
801 
802         var navItemClass = s.navItemClass;
803         var navItemSelectedClass = s.navItemSelectedClass;
804         var navItemHoverClass = s.navItemHoverClass;
805         var navDotsLeftClass = s.navDotsLeftClass;
806         var navDotsRightClass = s.navDotsRightClass;
807 
808         var nav_top_id = create_id(s.nav_top_id_prefix, container_id);
809         var nav_prev_id = create_id(s.nav_prev_id_prefix, container_id);
810         var nav_dots_left_id = create_id(s.nav_dots_left_id_prefix, container_id);
811         var nav_pages_id = create_id(s.nav_pages_id_prefix, container_id);
812         var nav_item_id_prefix = create_id(s.nav_item_id_prefix, container_id) + '_';
813         var nav_dots_right_id = create_id(s.nav_dots_right_id_prefix, container_id);
814         var nav_next_id = create_id(s.nav_next_id_prefix, container_id);
815         var nav_last_id = create_id(s.nav_last_id_prefix, container_id);
816 
817         var selector = '';
818         var i;
819         var nav_html = '';
820         var goto_page = parseInt(currentPage);
821 
822         var elem_nav_pages = $("#" + nav_pages_id);
823 
824         if(navPagesMode == 'continuous') {
825 
826             var nav_start, nav_end, mod, offset, totalSections;
827             nav_start = goto_page;
828 
829             // detect possible offset to navigation pages
830             if(totalPages < visiblePageLinks) {
831                 nav_start = 1;
832                 nav_end = totalPages;
833             } else {
834                 totalSections = Math.ceil(totalPages / visiblePageLinks);
835                 if(nav_start > visiblePageLinks * (totalSections - 1)) {
836                     nav_start = totalPages - visiblePageLinks + 1;
837                 } else {
838                     mod = nav_start % visiblePageLinks;
839                     if(mod == 0) {
840                         offset = -visiblePageLinks + 1;
841                     } else {
842                         offset = -mod + 1;
843                     }
844                     nav_start += offset;
845                 }
846                 nav_end = nav_start + visiblePageLinks - 1;
847             }
848 
849             // store nav_start nav_end
850             elem.data('nav_start', nav_start);
851             elem.data('nav_end', nav_end);
852 
853             // show - hide nav controls
854             if(showNavButtons) {
855                 var tmp = (showNavPages ? nav_start : goto_page);
856                 selector = "#" + nav_top_id + ', ' + "#" + nav_prev_id;
857                 if(tmp > 1) {
858                     $(selector).show();
859                 } else {
860                     $(selector).hide();
861                 }
862 
863                 tmp = (showNavPages ? nav_end : goto_page);
864                 selector = "#" + nav_next_id + ', ' + "#" + nav_last_id;
865                 if(tmp < totalPages) {
866                     $(selector).show();
867                 } else {
868                     $(selector).hide();
869                 }
870             }
871 
872             // create nav pages html
873             if(nav_start > 1) {
874                 nav_html += '<div id="' + nav_dots_left_id + '">...</div>';
875             }
876             for(i = nav_start; i <= nav_end; i++) {
877                 nav_html += '<div id="' + nav_item_id_prefix + i + '">' + i + '</div>';
878             }
879             if(nav_end < totalPages) {
880                 nav_html += '<div id="' + nav_dots_right_id + '">...</div>';
881             }
882             elem_nav_pages.html(nav_html);
883 
884 
885         } else if(navPagesMode == 'first-last-always-visible') {
886 
887             if(totalPages <= 5) {
888                 for(i = 1; i <= totalPages; i++) {
889                     nav_html += '<div id="' + nav_item_id_prefix + i + '">' + i + '</div>';
890                 }
891             } else {
892 
893                 var pageLinks = visiblePageLinks;
894                 if(pageLinks < 5) {
895                     pageLinks = 5;
896                 }
897                 if(pageLinks > totalPages) {
898                     pageLinks = totalPages;
899                 }
900 
901                 var median_links = pageLinks - 2;
902                 var left_links = Math.floor((median_links - 1) / 2);
903                 var right_links = Math.ceil((median_links - 1) / 2);
904 
905                 // create nav pages html
906                 if(goto_page <= median_links) {
907                     for(i = 1; i <= median_links + 1; i++) {
908                         nav_html += '<div id="' + nav_item_id_prefix + i + '">' + i + '</div>';
909                     }
910                     nav_html += '<div id="' + nav_dots_right_id + '">...</div>';
911                     nav_html += '<div id="' + nav_item_id_prefix + totalPages + '">' + totalPages + '</div>';
912                 } else if(goto_page > totalPages - median_links) {
913                     nav_html += '<div id="' + nav_item_id_prefix + '1' + '">1</div>';
914                     nav_html += '<div id="' + nav_dots_left_id + '">...</div>';
915                     for(i = totalPages - median_links; i <= totalPages; i++) {
916                         nav_html += '<div id="' + nav_item_id_prefix + i + '">' + i + '</div>';
917                     }
918                 } else {
919                     nav_html += '<div id="' + nav_item_id_prefix + '1' + '">1</div>';
920                     nav_html += '<div id="' + nav_dots_left_id + '">...</div>';
921                     for(i = goto_page - left_links; i <= goto_page + right_links; i++) {
922                         nav_html += '<div id="' + nav_item_id_prefix + i + '">' + i + '</div>';
923                     }
924                     nav_html += '<div id="' + nav_dots_right_id + '">...</div>';
925                     nav_html += '<div id="' + nav_item_id_prefix + totalPages + '">' + totalPages + '</div>';
926                 }
927 
928             }
929 
930             elem_nav_pages.html(nav_html);
931 
932             // show - hide nav controls
933             if(showNavButtons) {
934                 selector = "#" + nav_top_id + ', ' + "#" + nav_prev_id;
935                 if(goto_page > 1) {
936                     $(selector).show();
937                 } else {
938                     $(selector).hide();
939                 }
940 
941                 selector = "#" + nav_next_id + ', ' + "#" + nav_last_id;
942                 if(goto_page < totalPages) {
943                     $(selector).show();
944                 } else {
945                     $(selector).hide();
946                 }
947             }
948         }
949 
950         // apply style for navigation items (pages)
951         $("#" + nav_dots_left_id).removeClass().addClass(navDotsLeftClass);
952         $("#" + nav_dots_right_id).removeClass().addClass(navDotsRightClass);
953 
954         var elem_nav_items = $('[id^="' + nav_item_id_prefix + '"]');
955         elem_nav_items.removeClass().addClass(navItemClass);
956         $("#" + nav_item_id_prefix + goto_page).removeClass().addClass(navItemSelectedClass);
957 
958         if(navItemHoverClass != '') {
959             elem_nav_items.hover(
960                 function() {
961                     $(this).addClass(navItemHoverClass);
962                 },
963                 function() {
964                     $(this).removeClass(navItemHoverClass);
965                 }
966             );
967         }
968     };

973      * @param goto_page
974      * @param update_slider

976     var update_current_page = function(container_id, goto_page, update_slider) {
977

979         var elem = $("#" + container_id);
980         var s = $(elem).jui_pagination('getAllOptions');
981

983 
984         var showCurrentPage = s.showCurrentPage;
985 
986         var sliderElementID = s.sliderElementID;
987         var slider_id = (!sliderElementID ? create_id(s.slider_id_prefix, container_id) : sliderElementID);
988         var sliderOrientation = s.sliderOrientation;
989 
990         var current_id = create_id(s.nav_current_page_id_prefix, container_id);

998         $("#" + nav_item_id_prefix + goto_page).removeClass().addClass(navItemSelectedClass);
999 
1000         // update current page indicator
1001         if(showCurrentPage) {
1002             $("#" + current_id).text(goto_page);
1003         }
1004 
1005         // update slider if exists
1006         if(update_slider) {
1007             var elem_slider = $("#" + slider_id);
1008             if(typeof(elem_slider.data("slider")) == 'object') {
1009                 elem_slider.slider({
1010                     'value': (sliderOrientation == 'horizontal' ? goto_page : totalPages - goto_page + 1)
1011                 });
1012             }
1013         }
1014 
1015         // trigger event onChangePage
1016         elem.triggerHandler("onChangePage", goto_page);
